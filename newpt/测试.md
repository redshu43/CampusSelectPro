1. 这段代码会创建一个存储过程，自动往 stu 表里插入 1000 个账号（test_0 到 test_999），密码统一为 123456。
```mysql
-- 1. 如果之前有测试数据，建议先清空 (可选)
-- DELETE FROM stu WHERE account LIKE 'test_%';

-- 2. 修改结束符 (为了写存储过程)
DELIMITER //

-- 3. 创建存储过程
CREATE PROCEDURE batch_insert_students()
BEGIN
  DECLARE i INT DEFAULT 0;
  WHILE i < 1000 DO
    INSERT INTO stu (name, gender, age, account, password, role)
    VALUES (
      CONCAT('抢课机器人', i), -- 姓名
      1,                       -- 性别
      20,                      -- 年龄
      CONCAT('test_', i),      -- 账号: test_0, test_1...
      '$2a$10$N.zmdr9k7uOCQb376NoUnutj8iAt6.VwUvnuo7wcpScCnIsfbGVxK', -- 密码密文 (即 123456)
      'ROLE_STUDENT'           -- 角色
    );
    SET i = i + 1;
  END WHILE;
END //

-- 4. 恢复结束符
DELIMITER ;

-- 5. 执行存储过程 (生成数据)
CALL batch_insert_students();

-- 6. 用完删除存储过程 (清理垃圾)
DROP PROCEDURE batch_insert_students;
```
